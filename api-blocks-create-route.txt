// src/app/api/blocks/create/route.ts
import { NextResponse } from "next/server";
import { createClient, SupabaseClient } from "@supabase/supabase-js";

type CreateBlockPayload = {
  message_id: string;
  title: string;
  description?: string | null;
  start_at?: string | null;
  end_at?: string | null;
  timezone?: string | null;

  capacity_total?: number | null;
  capacity_remaining?: number | null;

  status?: "draft" | "live" | "paused" | "sold_out" | "expired" | string;
  action_type?: "book" | "reserve" | "enquire" | string;

  price_pence?: number | null;
  currency?: string | null;
  visibility?: "public" | "private" | "unlisted" | string;

  sort_rank?: number | null;
  meta?: Record<string, unknown> | null;
};

const ALLOWED_STATUS = new Set(["draft", "live", "paused", "sold_out", "expired"]);
const ALLOWED_ACTION = new Set(["book", "reserve", "enquire"]); // extend if you add more
const ALLOWED_VISIBILITY = new Set(["public", "private", "unlisted"]); // extend if you add more

function mustEnv(name: string): string {
  const v = process.env[name];
  if (!v) throw new Error(`Missing env: ${name}`);
  return v;
}

function supabaseAdmin(): SupabaseClient {
  const url =
    process.env.NEXT_PUBLIC_SUPABASE_URL ||
    process.env.SUPABASE_URL ||
    mustEnv("NEXT_PUBLIC_SUPABASE_URL");
  const key = process.env.SUPABASE_SERVICE_ROLE_KEY || mustEnv("SUPABASE_SERVICE_ROLE_KEY");

  return createClient(url, key, {
    auth: { persistSession: false, autoRefreshToken: false },
  });
}

function isIsoDate(s: unknown): s is string {
  if (typeof s !== "string") return false;
  const d = new Date(s);
  return !Number.isNaN(d.getTime());
}

function jsonError(status: number, body: Record<string, unknown>) {
  return NextResponse.json(body, { status });
}

function isConstraintViolation(err: any) {
  const code = String(err?.code || "");
  return code === "23P01" || code === "23505";
}

async function fetchConflictingLiveBlock(
  supabase: SupabaseClient,
  tagId: string,
  startIso: string | null,
  endIso: string | null
) {
  // With strict live windows, we assume startIso/endIso are non-null when status=live.
  // Still defensive in case of misuse.
  if (!startIso || !endIso) return null;

  // Overlap for [): existing.start < newEnd AND existing.end > newStart
  const { data, error } = await supabase
    .from("availability_blocks")
    .select("id, tag_id, status, start_at, end_at, capacity_total, capacity_remaining, created_at")
    .eq("tag_id", tagId)
    .eq("status", "live")
    .not("start_at", "is", null)
    .not("end_at", "is", null)
    .lt("start_at", endIso)
    .gt("end_at", startIso)
    .order("start_at", { ascending: true })
    .limit(1);

  if (error) return null;
  return data?.[0] ?? null;
}

export async function POST(req: Request) {
  try {
    const supabase = supabaseAdmin();

    let payload: CreateBlockPayload;
    try {
      payload = (await req.json()) as CreateBlockPayload;
    } catch {
      return jsonError(400, { ok: false, error: "INVALID_JSON" });
    }

    const message_id = (payload.message_id || "").trim();
    const title = (payload.title || "").trim();
    if (!message_id) return jsonError(400, { ok: false, error: "message_id is required" });
    if (!title) return jsonError(400, { ok: false, error: "title is required" });

    const start_at = payload.start_at ?? null;
    const end_at = payload.end_at ?? null;

    if (start_at !== null && start_at !== "" && !isIsoDate(start_at)) {
      return jsonError(400, { ok: false, error: "start_at must be ISO datetime" });
    }
    if (end_at !== null && end_at !== "" && !isIsoDate(end_at)) {
      return jsonError(400, { ok: false, error: "end_at must be ISO datetime" });
    }

    const startIso = start_at ? new Date(start_at).toISOString() : null;
    const endIso = end_at ? new Date(end_at).toISOString() : null;

    if (startIso && endIso) {
      const s = new Date(startIso).getTime();
      const e = new Date(endIso).getTime();
      if (e <= s) return jsonError(400, { ok: false, error: "end_at must be after start_at" });
    }

    const status = String(payload.status ?? "live").trim().toLowerCase();
    const action_type = String(payload.action_type ?? "book").trim().toLowerCase();
    const visibility = String(payload.visibility ?? "public").trim().toLowerCase();

    if (!ALLOWED_STATUS.has(status)) {
      return jsonError(400, { ok: false, error: "INVALID_STATUS", status });
    }
    if (!ALLOWED_ACTION.has(action_type)) {
      return jsonError(400, { ok: false, error: "INVALID_ACTION_TYPE", action_type });
    }
    if (!ALLOWED_VISIBILITY.has(visibility)) {
      return jsonError(400, { ok: false, error: "INVALID_VISIBILITY", visibility });
    }

    // Live requires a window (you already aligned the product to this)
    if (status === "live") {
      if (!startIso || !endIso) {
        return jsonError(400, {
          ok: false,
          error: "LIVE_REQUIRES_WINDOW",
          message: "Live blocks must include start_at and end_at (ISO).",
        });
      }
    }

    const capacity_total = payload.capacity_total ?? 1;
    const capacity_remaining =
      payload.capacity_remaining === null || payload.capacity_remaining === undefined
        ? null
        : payload.capacity_remaining;

    if (!Number.isFinite(capacity_total) || capacity_total < 1) {
      return jsonError(400, { ok: false, error: "capacity_total must be >= 1" });
    }
    if (capacity_remaining !== null) {
      if (!Number.isFinite(capacity_remaining)) {
        return jsonError(400, { ok: false, error: "capacity_remaining must be a number" });
      }
      if (capacity_remaining < 0 || capacity_remaining > capacity_total) {
        return jsonError(400, {
          ok: false,
          error: "capacity_remaining out of range",
          capacity_total,
          capacity_remaining,
        });
      }
    }

    const timezone = (payload.timezone ?? "Europe/London") || "Europe/London";
    const price_pence = payload.price_pence ?? 0;
    const currency = (payload.currency ?? "GBP") || "GBP";
    const sort_rank = payload.sort_rank ?? 0;
    const description = payload.description ?? null;

    // Owner lookup (canonical)
    const { data: msg, error: msgErr } = await supabase
      .from("messages")
      .select("id, user_id")
      .eq("id", message_id)
      .maybeSingle();

    if (msgErr) {
      return jsonError(500, { ok: false, error: "FAILED_TO_LOOKUP_MESSAGE", details: msgErr });
    }
    if (!msg?.id) {
      return jsonError(404, { ok: false, error: "MESSAGE_NOT_FOUND", message_id });
    }
    if (!msg.user_id) {
      return jsonError(400, { ok: false, error: "MESSAGE_HAS_NO_OWNER", message_id });
    }

    const meta = {
      ...(payload.meta ?? {}),
      created_via: "api/blocks/create",
      policy_version: 1,
    };

    // RPC call (atomic auto-pause + insert)
    const { data: rpcData, error: rpcErr } = await supabase.rpc("create_block_autopause", {
      p_message_id: message_id,
      p_title: title,
      p_owner_id: msg.user_id,

      p_description: description,
      p_start_at: startIso,
      p_end_at: endIso,
      p_timezone: timezone,

      p_capacity_total: capacity_total,
      p_capacity_remaining: capacity_remaining,

      p_status: status,
      p_action_type: action_type,
      p_visibility: visibility,

      p_price_pence: price_pence,
      p_currency: currency,

      p_sort_rank: sort_rank,
      p_meta: meta,
    });

    if (rpcErr) {
      const code = String((rpcErr as any)?.code || "");

      if (code === "42501") {
        return jsonError(403, {
          ok: false,
          error: "FORBIDDEN",
          postgres: { code, details: (rpcErr as any)?.details, hint: (rpcErr as any)?.hint },
          message: rpcErr.message,
        });
      }

      if (code === "23503") {
        return jsonError(404, {
          ok: false,
          error: "MESSAGE_NOT_FOUND",
          message_id,
          postgres: { code, details: (rpcErr as any)?.details, hint: (rpcErr as any)?.hint },
          message: rpcErr.message,
        });
      }

      if (isConstraintViolation(rpcErr)) {
        const conflict = await fetchConflictingLiveBlock(supabase, message_id, startIso, endIso);
        return jsonError(409, {
          ok: false,
          error: "OVERLAP_CONFLICT",
          message:
            "A live availability block overlaps an existing live block for this tag. Choose a different time or pause/expire the existing block.",
          conflict,
          postgres: { code, details: (rpcErr as any)?.details, hint: (rpcErr as any)?.hint },
        });
      }

      return jsonError(500, {
        ok: false,
        error: "RPC_FAILED",
        postgres: {
          code,
          details: (rpcErr as any)?.details,
          hint: (rpcErr as any)?.hint,
          message: rpcErr.message,
        },
      });
    }

    // Contract hardening: always include paused_ids, even if future function changes
    const ok = Boolean((rpcData as any)?.ok);
    const block = (rpcData as any)?.block ?? null;
    const paused_ids = Array.isArray((rpcData as any)?.paused_ids) ? (rpcData as any).paused_ids : [];

    return NextResponse.json({ ok, block, paused_ids });
  } catch (e: any) {
    return jsonError(500, { ok: false, error: "SERVER_ERROR", message: String(e?.message ?? e) });
  }
}
